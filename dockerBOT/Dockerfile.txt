# Используем официальный образ Python (нет необходимости в двух базовых образах)
FROM python:3.11-slim

# Устанавливаем системные зависимости
RUN apt-get update && \
    apt-get install -y --no-install-recommends gcc python3-dev && \
    rm -rf /var/lib/apt/lists/*

# Устанавливаем переменную окружения для корректного вывода
ENV PYTHONUNBUFFERED=1

# Определяем рабочую директорию
WORKDIR /app

# Копируем и устанавливаем зависимости
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Добавляем необходимые библиотеки, если их нет в requirements.txt
RUN pip install telebot python-telegram-bot

# Копируем исходный код
COPY . .

# Указываем команду для запуска
CMD ["python", "TQW.py"]
